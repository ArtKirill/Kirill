#include <time.h>
#include <stdio.h>
#define N 5
 
void summa()
{
    int sum;
    sum = 0;
    for(int i = 0; i < N; i++)
    { 
        for (int j = 0; j < N; j++)
        { 
            if ( matr[i][j] > 0 )
            sum+= matr[i][j];
        }
        printf("\n\nСумма положительных: %i", sum);
    }
}
void kolotr()
{
    int otr;
    otr = 0;
    for( int i = 0; i < N; i++)
    { 
	for(int j = 0; j < N; j++)
        {
	    if ( matr[i][j] < 0 )
            otr++;
        }
        printf("\n\nколичество отрицательных: %i\n", otr);
    }
}
